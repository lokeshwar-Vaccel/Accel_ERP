## Usage

### Running the Seed Script

To create this seed data, run:

```bash
cd api
npx ts-node --transpile-only -r dotenv/config src/scripts/complete-seed.ts
```

### Data Verification

To verify the created data:

```bash
cd api
npx ts-node --transpile-only -r dotenv/config -e "
import { connectDB } from './src/database/connection';
import { Customer } from './src/models/Customer';
import { Product } from './src/models/Product';
import { Stock } from './src/models/Stock';
import mongoose from 'mongoose';

async function verifyData() {
  await connectDB();
  
  const suppliers = await Customer.find({ type: 'supplier' });
  const products = await Product.find({});
  const stockEntries = await Stock.find({}).populate('product', 'name partNo');
  
  console.log('📊 Data Verification:');
  console.log(\`Suppliers: \${suppliers.length}\`);
  suppliers.forEach(s => console.log(\`  - \${s.name} (\${s.email})\`));
  
  console.log(\`\\nProducts: \${products.length}\`);
  products.forEach(p => console.log(\`  - \${p.name} (\${p.partNo}) - Qty: \${p.quantity}\`));
  
  console.log(\`\\nStock Entries: \${stockEntries.length}\`);
  stockEntries.forEach(s => console.log(\`  - \${s.product?.name} - Qty: \${s.quantity}\`));
  
  await mongoose.disconnect();
}
verifyData();
"
```
